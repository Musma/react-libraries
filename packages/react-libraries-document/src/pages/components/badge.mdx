import { FillAddBoxIcon } from "@musma/react-icons";

# Badge

공통으로 사용 될 Badge 디자인입니다.
배지는 일반적으로 읽지 않은 메시지 수를 표시하는 눈길을 끄는 표현으로 알림 또는 사용자 아바타 근처에 표시됩니다.
카운트가 0이면 배지가 숨겨지고, 1 이상 일 경우 표시됩니다.

### 사용방법

```
import { Badge } from '@musma/react-component'

<Badge>
  <SomethingIcon />
</Badge>
```

<div className="flex gap-10 rounded border p-10">
  <Badge badgeCount={1}>
    <FillAddBoxIcon />
  </Badge>
  <Badge badgeCount={10}>
    <FillAddBoxIcon />
  </Badge>
  <Badge badgeCount={20009}>
    <FillAddBoxIcon />
  </Badge>
</div>

export default ({ children }) => (
