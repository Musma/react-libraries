import { useState } from "react";
import { Modal, Button, useModalManager } from "@musma/react-component";

import { Box } from "src/components";

export const ModalDemo = () => {
  const manager = useModalManager();
  const [isOpen, setIsOpen] = useState(false);
  const [isOpen2, setIsOpen2] = useState(false);

return (

<div>
  <Button label={"Open Modal"} onClick={() => setIsOpen(true)} />
  <Modal
    title={"Modal"}
    isOpen={isOpen}
    onClose={() => setIsOpen(false)}
    closeOnEscPress={true}
    modalManager={manager}
    buttonOption={{
      label: "Button",
      onClick: () => undefined,
      secondLabel: "Button",
      onSecondClick: () => undefined,
    }}
  >
    <div className="flex h-full w-full flex-col items-center justify-center">
      I'm Modal
      <Button label={"Open Nested Modal"} onClick={() => setIsOpen2(true)} />
    </div>
  </Modal>
  <Modal
    title={"Nested Modal"}
    isOpen={isOpen2}
    onClose={() => setIsOpen2(false)}
    closeOnEscPress={true}
    modalManager={manager}
    buttonOption={{
      label: "Button",
      onClick: () => undefined,
      secondLabel: "Button",
      onSecondClick: () => undefined,
    }}
  >
    <div className="flex h-full w-full flex-col items-center justify-center">
      I'm nested Modal
    </div>
  </Modal>
</div>
); };

# Modal

공통으로 사용 될 Modal 디자인입니다.

### 사용방법

```
import { Modal } from '@musma/react-component'

<Modal title="title">
  <SomethingIcon />
</Modal>
```
